# AIM_Pedagogical_Stages

pedagogical_stages:

  initiation:
    purpose: establish ritual and teacher-student field
    example_texts:
      - Taittirīya Śikṣāvallī 1
      - Taittirīya Śikṣāvallī 2

  symbolic_training:
    purpose: train symbolic and cosmological cognition
    example_texts:
      - Śikṣāvallī 3–7

  mantric_unification:
    purpose: collapse symbolic multiplicity into a single sacred sign
    example_texts:
      - Śikṣāvallī 8

  ethical_stabilization:
    purpose: stabilize character before deeper inquiry
    example_texts:
      - Śikṣāvallī 9–11

  pedagogical_closure:
    purpose: close transmission and seal instruction
    example_texts:
      - Śikṣāvallī 12
